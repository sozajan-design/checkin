                        setGuests(newGuests);

                        setScanResult({ type: 'success', message: '報到成功', data: newGuests[guestIndex], color: 'green' });

                    <nav className="bg-white border-t border-gray-100 pb-safe shadow-inner z-40 no-print">

                        <div className="flex justify-around items-center h-20">

                            <button onClick={() => setActiveTab('scan')} className={`flex flex-col items-center justify-center w-full h-full space-y-1 transition-colors ${activeTab === 'scan' ? 'text-indigo-600' : 'text-gray-300'}`}>

                                <Icon name="Camera" size={28} /> <span className="text-[10px] font-black uppercase">Scan</span>

                            </button>

                            <button onClick={() => { if(isAdmin) setActiveTab('list'); else setShowLoginModal(true); }} className={`flex flex-col items-center justify-center w-full h-full space-y-1 transition-colors ${activeTab === 'list' ? 'text-indigo-600' : 'text-gray-300'}`}>

                                <Icon name={isAdmin ? "User" : "Lock"} size={28} /> <span className="text-[10px] font-black uppercase">List</span>

                            </button>

                            <button onClick={() => { if(isAdmin) setActiveTab('settings'); else setShowLoginModal(true); }} className={`flex flex-col items-center justify-center w-full h-full space-y-1 transition-colors ${activeTab === 'settings' ? 'text-indigo-600' : 'text-gray-300'}`}>

                                <Icon name={isAdmin ? "Database" : "Lock"} size={28} /> <span className="text-[10px] font-black uppercase">Data</span>

                            </button>

                        </div>

                    </nav>


                    {showLoginModal && (

                        <div className="fixed inset-0 bg-black/80 backdrop-blur-md z-50 flex items-center justify-center p-6">

                            <div className="bg-white w-full max-w-sm rounded-[2.5rem] p-10 shadow-2xl animate-in zoom-in duration-300">

                                <h3 className="text-2xl font-black text-gray-800 text-center mb-8 uppercase">Login</h3>

                                <div className="grid grid-cols-3 gap-4 mb-10 text-xl font-black">

                                    {[1,2,3,4,5,6,7,8,9, '✕', 0, '⌫'].map(v => (

                                        <button key={v} onClick={() => {

                                            if(v === '✕') setShowLoginModal(false);

                                            else if(v === '⌫') setPasswordInput(p => p.slice(0,-1));

                                            else if(passwordInput.length < 4) setPasswordInput(p => p + v);

                                        }} className="h-14 rounded-2xl bg-gray-50 active:bg-indigo-600 active:text-white transition-all">{v}</button>

                                    ))}

                                </div>

                                <button onClick={() => {

                                    if(passwordInput === ADMIN_PASSWORD) { setIsAdmin(true); setShowLoginModal(false); setActiveTab('list'); }

                                    else { setPasswordInput(''); alert('密碼錯誤'); }

                                }} className="w-full py-4 bg-indigo-600 text-white rounded-2xl font-bold shadow-lg">進入管理模式</button>

                                <div className="mt-8 text-center text-gray-300 text-[10px] font-bold tracking-widest">{COMPANY_NAME_ZH} • v{APP_VERSION}</div>

                            </div>

                        </div>

                    )}


                    {showManualInput && (

                        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm z-50 flex items-end sm:items-center justify-center no-print">

                            <div className="bg-white w-full max-w-md rounded-t-[2rem] sm:rounded-[2rem] p-8 shadow-2xl animate-in slide-in-from-bottom duration-300">

                                <div className="flex justify-between items-center mb-8">

                                    <h3 className="text-2xl font-black">手動輸入票號</h3>

                                    <button onClick={() => setShowManualInput(false)} className="bg-gray-100 p-2 rounded-full active:scale-90 transition-transform"><Icon name="X" size={20}/></button>

                                </div>

                                <input autoFocus type="text" placeholder="例: A001" className="w-full text-3xl font-black p-5 bg-gray-50 border-2 border-indigo-100 rounded-2xl focus:border-indigo-500 outline-none mb-8 text-center uppercase tracking-widest" onKeyDown={(e) => { if(e.key === 'Enter') { handleScanSuccess(e.target.value); setShowManualInput(false); } }} />

                                <button onClick={() => { const val = document.querySelector('input[type="text"]').value; handleScanSuccess(val); setShowManualInput(false); }} className="w-full bg-indigo-600 text-white py-5 rounded-2xl font-black text-xl shadow-lg active:scale-95 transition-transform text-center uppercase">OK</button>

                            </div>

                        </div>

                    )}

                </div>

            );

        }


        function ScannerComponent({ onScan }) {

            useEffect(() => {

                const scanner = new Html5QrcodeScanner("reader", { 

                    fps: 20, 

                    qrbox: (viewWidth, viewHeight) => {

                        const boxSize = Math.floor(Math.min(viewWidth, viewHeight) * 0.75);

                        return { width: boxSize, height: boxSize };

                    },

                    aspectRatio: 1.0, 

                    showTorchButtonIfSupported: true,

                    rememberLastUsedCamera: true

                }, false);

                scanner.render((text) => {

                    scanner.clear().then(() => onScan(text));

                }, () => {});

                return () => scanner.clear().catch(()=>{});

            }, [onScan]);

            return <div id="reader" className="w-full h-full bg-black"></div>;

        }


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
